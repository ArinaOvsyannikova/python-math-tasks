n = int(input("Введите кол-во слагаемых: "))
s = 0

for i in range(1, n + 1):
    if i > 1:
        print("+", end="")
    else:
        print(end="")

    c = 1
    for j in range(i, 2 * i+1):
        if j != 2 * i:
            print(j, end="*")
        else:
            print(j, end="")
        c *= j

    s += c

print("=", s)
